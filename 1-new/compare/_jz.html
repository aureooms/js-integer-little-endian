Returns true if number is 0.

Parameters:

  • a must be an array.
    (first operand)

  • ai must be an int.
    (a left)

  • aj must be an int.
    (a right)

const _jz = function ( a , ai , aj ) { for ( ; ai < aj ; ++ai ) if ( a[ai] !== 0 ) return false ; return true ; } ; exports._jz = _jz ;
