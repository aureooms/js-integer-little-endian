Subtracts two big endian arrays, k >= i >= j wraps

Parameters:

  • r must be an int.
    (base (radix))

  • a must be an array.
    (first operand)

  • ai must be an int.
    (a left)

  • aj must be an int.
    (a right)

  • b must be an array.
    (second operand)

  • bi must be an int.
    (b left)

  • bj must be an int.
    (b right)

  • c must be an array.
    (result, must be 0 initialized)

  • ci must be an int.
    (c left)

  • cj must be an int.
    (c right)

const _sub = function ( r , a, ai, aj, b, bi, bj, c, ci, cj){ var T, C = 0; while(--bj >= bi){ --aj; --cj; T = C; C = a[aj] < b[bj] + T; c[cj] = a[aj] - b[bj] + (C*r - T); } while(--aj >= ai){ --cj; T = C; C = a[aj] < T; c[cj] = a[aj] + (C*r - T); } if(C){ while(--cj >= ci){ c[cj] = r - 1; } } }; exports._sub = _sub;
